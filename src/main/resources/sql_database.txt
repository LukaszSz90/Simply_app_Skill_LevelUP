--How the database and tables for the application were created

CREATE DATABASE SKILL_APP;

USE SKILL_APP;

CREATE TABLE user (
	ID INT AUTO_INCREMENT NOT NULL,
    NAME VARCHAR(20) NOT NULL,
    PASSWORD VARCHAR(20) NOT NULL,
    REGISTER_DATE DATE NOT NULL,
    PRIMARY KEY (ID)
    );

CREATE TABLE skill (
	ID INT AUTO_INCREMENT NOT NULL,
    SKILL_NAME VARCHAR(40) UNIQUE NOT NULL,
    SKILL_LEVEL VARCHAR(40) NOT NULL,
    USER_EXP INT NOT NULL,
	NEXT_LEVEL_EXP INT NOT NULL,
    START_DATE DATE NOT NULL,
    LAST_LOGIN DATE NOT NULL,
    SERIES_DAY INT NOT NULL,
    USER_ID INT NOT NULL,
    PRIMARY KEY (ID),
    CONSTRAINT fk_user
		FOREIGN KEY (USER_ID)
			REFERENCES USER (ID)
);
